'₭ ' + value.toLocaleString() : value; }, anchor: type === 'bar' ? 'end' : 'center', align: type === 'bar' ? 'bottom' : 'center', color: '#fff' } } } }); }

    function filterStockHistory() {
      let dateVal = document.getElementById('stockSearchDate').value;
      let textVal = document.getElementById('stockSearchText').value.toLowerCase();
      
      let filtered = globalStockHistory.filter(h => {
        let matchDate = true;
        if (dateVal) {
          let hDateIso = new Date(h.date - (new Date().getTimezoneOffset() * 60000)).toISOString().split('T')[0];
          matchDate = (hDateIso === dateVal);
        }
        let matchText = true;
        if (textVal) {
          matchText = h.name.toLowerCase().includes(textVal) || (h.note && h.note.toLowerCase().includes(textVal));
        }
        return matchDate && matchText;
      });
      renderStockHistory(filtered);
    }

    function exportHistoryData(type) {
      let d = document.getElementById('searchDate').value;

      if(filtered.length === 0) { Swal.fire('ແຈ້ງເຕືອນ', 'ບໍ່ມີຂໍ້ມູນສຳລັບດາວໂຫຼດ', 'warning'); return; }
      let exportArr = filtered.map(o => ({ "Order_ID": o.orderId, "Date_Time": new Date(o.dateTime).toLocaleString('en-GB'), "Patient_ID": o.patientId, "Patient_Name": o.patientName, "Lab_Destination": o.labDest || 'In-house', "Tests": o.tests.map(t=>t.name).join(', '), "Total_Kip": o.totalPrice, "Status": o.status }));
      let fileName = "Order_History_" + new Date().toISOString().split('T')[0];

    }

    function loadParamSetupData() {
      google.script.run.withSuccessHandler(params => {
          if ($.fn.DataTable.isDataTable('#paramTable')) { $('#paramTable').DataTable().clear().destroy(); }
          let tbody = document.getElementById('paramTableBody'); if(!tbody) return;
          tbody.innerHTML = '';
          params.forEach(p => {
              let rangeText = '-';
              if(p.inputType === 'Number') { rangeText = `${p.min || 0} - ${p.max || 0}`; } 
              else if(p.inputType === 'Dropdown') { rangeText = `<small class="text-muted">${p.options}</small>`; }
              let typeBadge = p.inputType === 'Number' ? '<span class="badge bg-info text-dark">Number</span>' : (p.inputType === 'Dropdown' ? '<span class="badge bg-warning text-dark">Dropdown</span>' : '<span class="badge bg-secondary">Text</span>');
              tbody.innerHTML += `<tr>
                  <td class="fw-bold text-primary">${p.testName}</td><td class="fw-bold">${p.paramName}</td><td>${typeBadge}</td><td>${rangeText}</td><td>${p.unit || '-'}</td>
                  <td class="text-center"><button class="btn btn-sm btn-outline-danger py-0" onclick="deleteParameter(${p.rowIdx})" title="ລຶບ"><i class="bi bi-trash"></i></button></td>
              </tr>`;
          });
          initDT('paramTable', '450px');
      }).getTestParameters();
    }

    function toggleParamInputFields() {
      const type = document.getElementById('spInputType').value;

      document.getElementById('divDropdownSetup').style.display = (type === 'Dropdown') ? 'flex' : 'none';
    }

    function saveParameter() {
      const tName = document.getElementById('spTestName').value; const pName = document.getElementById('spParamName').value.trim();
      const iType = document.getElementById('spInputType').value; const pUnit = document.getElementById('spUnit').value.trim();
      let pMin = "", pMax = "", pOpt = "";
      if(!tName || !pName) { Swal.fire('ແຈ້ງເຕືອນ', 'ກະລຸນາເລືອກລາຍການກວດ ແລະ ໃສ່ຊື່ Parameter!', 'warning'); return; }
      if (iType === 'Number') { pMin = document.getElementById('spMin').value; pMax = document.getElementById('spMax').value; } 
      else if (iType === 'Dropdown') { pOpt = document.getElementById('spOptions').value.trim(); if(!pOpt) { Swal.fire('ແຈ້ງເຕືອນ', 'ກະລຸນາປ້ອນຕົວເລືອກ (Options)!', 'warning'); return; } }

      const payload = { testName: tName, paramName: pName, inputType: iType, unit: pUnit, min: pMin, max: pMax, options: pOpt };
      google.script.run.withSuccessHandler(res => {
          if(res.success) {
              Swal.fire('ສຳເລັດ', res.message, 'success');
              document.getElementById('spParamName').value = ''; document.getElementById('spMin').value = ''; document.getElementById('spMax').value = ''; document.getElementById('spOptions').value = '';
              loadParamSetupData();
          } else { Swal.fire('ຜິດພາດ', res.message, 'error'); }
      }).saveTestParameter(payload, sessionStorage.getItem('lis_username'));
    }

    function deleteParameter(rowIdx) {
      Swal.fire({ title: 'ລຶບການຕັ້ງຄ່ານີ້?', icon: 'warning', showCancelButton: true, confirmButtonColor: '#d33' }).then((result) => {

          }
      });
    }

    function applyFastEntry(testName, inputEl) {
        let values = inputEl.value.trim().split(/[\s,]+/);
        let inputs = document.querySelectorAll(`.result-input[data-test="${testName}"]`);
        let valIdx = 0;
        inputs.forEach(inp => {
            if (valIdx < values.length && inp.tagName === 'INPUT') { 
                inp.value = values[valIdx];
                valIdx++;
            }
        });
        inputEl.value = ''; 
    }

    function openResultModal(orderId) {
        if (!resultModalInstance) { resultModalInstance = new bootstrap.Modal(document.getElementById('resultEntryModal')); }
        document.getElementById('resOrderIdDisplay').innerText = orderId;
        document.getElementById('currentResultOrderId').value = orderId;
        let container = document.getElementById('resultFormContainer');
        container.innerHTML = '<div class="text-center py-4"><div class="spinner-border text-success"></div> ກຳລັງໂຫຼດຟອມ...</div>';
        resultModalInstance.show();

        google.script.run.withSuccessHandler(res => {
            if(!res.success) { container.innerHTML = `<div class="alert alert-danger">${res.message}</div>`; return; }
            let tests = res.tests;
            if(Object.keys(tests).length === 0) {
                container.innerHTML = `<div class="alert alert-warning"><i class="bi bi-exclamation-triangle"></i> ຍັງບໍ່ໄດ້ຕັ້ງຄ່າ Parameter ສຳລັບລາຍການກວດໃນບິນນີ້. <br>ກະລຸນາໄປທີ່ເມນູ <b>"ຕັ້ງຄ່າຜົນກວດ (Parameters)"</b>.</div>`;
                return;
            }

            let html = '';
            for (let testName in tests) {
                html += `<div class="mb-4 border rounded p-3 bg-white shadow-sm">`;
                html += `<div class="d-flex flex-wrap justify-content-between align-items-center border-bottom pb-2 mb-3 gap-2">
                            <h6 class="fw-bold text-primary m-0"><i class="bi bi-droplet-half me-2"></i>${testName}</h6>
                            <input type="text" class="form-control form-control-sm border-info" style="width:250px;" placeholder="Fast Entry (ພິມລວດດຽວແລ້ວຍະຫວ່າງ)" onkeyup="if(event.key==='Enter') applyFastEntry('${testName}', this)">
                         </div>`;
                html += `<div class="row g-2">`;
                
                let params = tests[testName];
                params.forEach(p => {
                    html += `<div class="col-md-6 d-flex align-items-center mb-2">`;
                    html += `<label class="form-label mb-0 fw-bold small text-muted" style="width: 140px;">${p.paramName}</label>`;
                    
                    let currentVal = p.savedValue !== undefined ? p.savedValue : '';

                    if (p.inputType === 'Dropdown') {
                        let opts = p.options ? p.options.split(',') : [];
                        html += `<select class="form-select form-select-sm border-primary result-input" data-test="${testName}" data-param="${p.paramName}">`;
                        html += `<option value="" disabled ${currentVal === '' ? 'selected' : ''}>-- ເລືອກ --</option>`;
                        opts.forEach(opt => { 
                            let isSelected = (opt.trim() == currentVal.toString().trim()) ? 'selected' : '';
                            html += `<option value="${opt.trim()}" ${isSelected}>${opt.trim()}</option>`; 
                        });
                        html += `</select>`;
                    } else {
                        html += `<input type="text" class="form-control form-control-sm border-primary result-input" data-test="${testName}" data-param="${p.paramName}" placeholder="ປ້ອນຜົນ..." value="${currentVal}">`;
                    }

                    let normalTxt = '-';
                    if (p.inputType === 'Number') { if(p.normalMin !== "" && p.normalMax !== "") normalTxt = `${p.normalMin} - ${p.normalMax}`; }

                    html += `<span class="ms-2 small text-muted w-50 text-truncate" title="${normalTxt}">${p.unit || ''} <br><span style="font-size:0.7rem;">(${normalTxt})</span></span>`;
                    html += `</div>`;
                });
                html += `</div></div>`;
            }
            container.innerHTML = html;
        }).getParametersForOrder(orderId);
    }

    function submitLabResults() {
        let orderId = document.getElementById('currentResultOrderId').value;
        let inputs = document.querySelectorAll('.result-input');
        let results = [];
        let hasEmpty = false;

        inputs.forEach(input => {
            let val = input.value.trim();
            if (!val) hasEmpty = true;
            results.push({ testName: input.getAttribute('data-test'), paramName: input.getAttribute('data-param'), value: val });
        });

        if (results.length === 0) return;
        if (hasEmpty) {
            Swal.fire({ title: 'ມີບາງຊ່ອງຍັງບໍ່ໄດ້ປ້ອນ', text: "ທ່ານຕ້ອງການບັນທຶກຜົນເທົ່າທີ່ມີກ່ອນບໍ່?", icon: 'warning', showCancelButton: true, confirmButtonText: 'ບັນທຶກເລີຍ', cancelButtonText: 'ກັບໄປປ້ອນຕໍ່' })
                .then((result) => { if (result.isConfirmed) { executeSaveResult(orderId, results); } });
        } else {
            executeSaveResult(orderId, results);
        }
    }

    function executeSaveResult(orderId, results) {
        let btn = document.getElementById('btnSaveResult');
        btn.innerHTML = '<span class="spinner-border spinner-border-sm"></span> ບັນທຶກ...'; btn.disabled = true;

        google.script.run.withSuccessHandler(res => {
            btn.innerHTML = '<i class="bi bi-check-circle me-1"></i> ບັນທຶກຜົນກວດ'; btn.disabled = false;
            if(res.success) {
                Swal.fire('ສຳເລັດ', res.message, 'success');
                resultModalInstance.hide();
                loadTable(); 
            } else { Swal.fire('ຜິດພາດ', res.message, 'error'); }
        }).saveLabResults(orderId, results, sessionStorage.getItem('lis_username'));
    }
