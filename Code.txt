function initDatabaseSheets() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // ກວດສອບ ແລະ ສ້າງ Sheet "Test_Parameters"
  var paramSheet = ss.getSheetByName('Test_Parameters');
  if (!paramSheet) {
    paramSheet = ss.insertSheet('Test_Parameters');
    paramSheet.appendRow(['Test_Name', 'Parameter_Name', 'Input_Type', 'Options', 'Unit', 'Normal_Min', 'Normal_Max']);
    paramSheet.getRange("A1:G1").setFontWeight("bold").setBackground("#D9EAD3");
  }
  
  // ກວດສອບ ແລະ ສ້າງ Sheet "Test_Results"
  var resSheet = ss.getSheetByName('Test_Results');
  if (!resSheet) {
    resSheet = ss.insertSheet('Test_Results');
    resSheet.appendRow(['Order_ID', 'Test_Name', 'Parameter_Name', 'Result_Value', 'Flag', 'Timestamp', 'User']);
    resSheet.getRange("A1:G1").setFontWeight("bold").setBackground("#D9EAD3");
  }
}

}

function loginUser(username, password) { 
  try { 
    var data = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Users').getDataRange().getValues(); 
    var inputUser = String(username).trim(); var inputPass = String(password).trim(); 
    for (var i = 1; i < data.length; i++) { 
      if (String(data[i][0]).trim() === inputUser && String(data[i][1]).trim() === inputPass) { 
        logActivity(inputUser, "Login", "System", "ເຂົ້າສູ່ລະບົບ"); 
        return { success: true, username: inputUser, role: data[i][2] }; 
      } 
    } 
    return { success: false, message: "Username ຫຼື Password ບໍ່ຖືກຕ້ອງ!" }; 
  } catch(e) { return { success: false, message: e.toString() }; } 
}

function recordStockTransaction(id, name, type, qty, note, user) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var numQty = Number(qty); ss.getSheetByName('Stock_Transactions').appendRow([new Date(), id, name, type, numQty, note, user]); if (type === "OUT") { var invSheet = ss.getSheetByName('Inventory_Lots'); if (invSheet) { var invData = invSheet.getDataRange().getValues(); var lotsToDeduct = []; for (var r = 1; r < invData.length; r++) { if (invData[r][1] === id && Number(invData[r][9]) > 0) { lotsToDeduct.push({ rowIndex: r + 1, expDate: invData[r][7] ? new Date(invData[r][7]).getTime() : 9999999999999, currentQty: Number(invData[r][9]) }); } } lotsToDeduct.sort(function(a, b) { return a.expDate - b.expDate; }); var remainingToDeduct = numQty; for (var k = 0; k < lotsToDeduct.length; k++) { if (remainingToDeduct <= 0) break; var lot = lotsToDeduct[k]; var deductAmount = Math.min(lot.currentQty, remainingToDeduct); invSheet.getRange(lot.rowIndex, 10).setValue(lot.currentQty - deductAmount); remainingToDeduct -= deductAmount; } } } return { success: true, message: "ບັນທຶກສຳເລັດ! ຕັດສະຕັອກແລ້ວ." }; } catch(e) { return { success: false, message: e.toString() }; } }

function submitTestOrder(data) { 
  try { 
    var ss = SpreadsheetApp.getActiveSpreadsheet(); 
    var sheet = ss.getSheetByName('Test_Orders'); 
    var orderId = data.existingOrderId; 
    var isEdit = false; 
    
    if(orderId) { 
      isEdit = true; 
      var allData = sheet.getDataRange().getValues(); 
      for (var i = allData.length - 1; i > 0; i--) { 
        if (allData[i][0] === orderId) { sheet.deleteRow(i + 1); } 
      } 
    } else { 
      var dateStr = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyyMMdd"); 
      var randomStr = Math.floor(Math.random() * 10000).toString().padStart(4, '0'); 
      orderId = "ORD-" + dateStr + "-" + randomStr; 
    } 
    
    var rowsToInsert = []; 
    var cart = data.cart; 
    for (var i = 0; i < cart.length; i++) { 
      var item = cart[i]; 
      var rowData = [ orderId, data.orderDateTime, data.timeSlot, data.visitType, data.insite, data.patientId, data.patientName, data.age, data.gender, data.doctor, data.department, data.testType, item.name, item.price, data.totalPrice, data.totalPrice, data.labDest, data.sender, "Pending", item.category ]; 
      rowsToInsert.push(rowData); 
    } 
    
    if (rowsToInsert.length > 0) {
      sheet.getRange(sheet.getLastRow() + 1, 1, rowsToInsert.length, rowsToInsert[0].length).setValues(rowsToInsert); 
    }
    logActivity(data.loggedUser, isEdit ? "Edit Order" : "Create Order", orderId, "Total: ₭" + data.totalPrice); 

    // ---- Auto Deduct Reagents (ຕັດສະຕັອກອັດຕະໂນມັດ) ----
    if (!isEdit) { 
      try {
        var mappingSheet = ss.getSheetByName('Test_Reagent_Mapping');
        if (mappingSheet) {
          var mappingData = mappingSheet.getDataRange().getValues();
          var testReagentMap = {};
          
          for (var m = 1; m < mappingData.length; m++) {
            if (mappingData[m][0]) {
              var tName = String(mappingData[m][0]).trim().toLowerCase();
              if (!testReagentMap[tName]) testReagentMap[tName] = [];
              testReagentMap[tName].push({ 
                id: mappingData[m][1], name: mappingData[m][2], qty: Number(mappingData[m][3]) 
              });
            }
          }

          var groupedDeductions = {};
          for (var c = 0; c < cart.length; c++) {
            if (cart[c].name) {
              var cName = String(cart[c].name).trim().toLowerCase();
              var reqs = testReagentMap[cName];
              if (reqs) {
                for (var r = 0; r < reqs.length; r++) {
                  var rId = reqs[r].id;
                  if(!groupedDeductions[rId]) {
                     groupedDeductions[rId] = {id: rId, name: reqs[r].name, qty: 0};
                  }
                  groupedDeductions[rId].qty += reqs[r].qty;
                }
              }
            }
          }

          for (var key in groupedDeductions) {
            var dItem = groupedDeductions[key];
            if(dItem.qty > 0) { recordStockTransaction(dItem.id, dItem.name, "OUT", dItem.qty, "Auto-deduct ຈາກບິນ " + orderId, data.loggedUser); }
          }
        }
      } catch(err) {
        logActivity("System", "Auto-Deduct Error", orderId, err.toString());
      }
    }
    return { success: true, message: "ບັນທຶກ Order " + orderId + " ສຳເລັດ!" }; 
  } catch (error) { return { success: false, message: error.toString() }; } 
}

function getParametersForOrder(orderId) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var orderData = ss.getSheetByName('Test_Orders').getDataRange().getValues();
    var orderedTests = [];
    for (var i = 1; i < orderData.length; i++) {
      if (String(orderData[i][0]).trim() === String(orderId).trim() && orderData[i][18] !== 'Cancelled') { 
        orderedTests.push(String(orderData[i][12]).trim()); 
      }
    }
    
    // ດຶງຜົນກວດເກົ່າທີ່ເຄີຍບັນທຶກແລ້ວ ມາປຽບທຽບ (ໃຊ້ .trim() ເພື່ອໃຫ້ແນ່ນອນ)
    var resSheet = ss.getSheetByName('Test_Results');
    var existingResults = {};
    if (resSheet) {
       var resData = resSheet.getDataRange().getValues();
       for (var r = 1; r < resData.length; r++) {
          if (String(resData[r][0]).trim() === String(orderId).trim()) {
             var k = String(resData[r][1]).trim() + "_" + String(resData[r][2]).trim(); 
             existingResults[k] = resData[r][3]; 
          }
       }
    }

    var paramData = ss.getSheetByName('Test_Parameters').getDataRange().getValues();
    var formStructure = {};
    for (var j = 1; j < paramData.length; j++) {
      var tName = String(paramData[j][0]).trim();
      if (orderedTests.indexOf(tName) !== -1) {
        if (!formStructure[tName]) formStructure[tName] = [];
        
        var pName = String(paramData[j][1]).trim();
        var pKey = tName + "_" + pName;
        var savedVal = existingResults[pKey] !== undefined ? existingResults[pKey] : ""; 

        formStructure[tName].push({
          paramName: pName, 
          inputType: paramData[j][2], 
          options: paramData[j][3], 
          unit: paramData[j][4], 
          normalMin: paramData[j][5], 
          normalMax: paramData[j][6],
          savedValue: savedVal 
        });
      }
    }
    return { success: true, orderId: orderId, tests: formStructure };
  } catch(e) { return { success: false, message: e.toString() }; }
}

function saveLabResults(orderId, results, user) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var resSheet = ss.getSheetByName('Test_Results');
    var paramData = ss.getSheetByName('Test_Parameters').getDataRange().getValues();
    var timestamp = new Date();
    
    // ສ້າງ Map ເພື່ອກວດສອບຄ່າ Min/Max 
    var limits = {};
    for(var p=1; p<paramData.length; p++) {
       limits[String(paramData[p][0]).trim() + "_" + String(paramData[p][1]).trim()] = {min: paramData[p][5], max: paramData[p][6]};
    }

    // --- ວິທີໃໝ່: ລຶບຜົນກວດເກົ່າຂອງບິນນີ້ຖິ້ມທັງໝົດ ກ່ອນຈະບັນທຶກໃໝ່ (ປ້ອງກັນຊ້ຳ 100%) ---
    var resData = resSheet.getDataRange().getValues();
    var rowsToDelete = [];
    for (var r = resData.length - 1; r > 0; r--) { // ຕ້ອງລຶບຈາກລຸ່ມຂຶ້ນເທິງ
       if (String(resData[r][0]).trim() === String(orderId).trim()) {
          rowsToDelete.push(r + 1);
       }
    }
    for (var d = 0; d < rowsToDelete.length; d++) {
       resSheet.deleteRow(rowsToDelete[d]);
    }
    // --------------------------------------------------------------------

    // ກຽມຂໍ້ມູນໃໝ່ເພື່ອບັນທຶກ
    var rowsToInsert = [];
    for (var i = 0; i < results.length; i++) {
      var tName = String(results[i].testName).trim();
      var pName = String(results[i].paramName).trim();
      var val = results[i].value;
      var flag = "Normal";

      var limit = limits[tName + "_" + pName];
      if (limit && val !== "") {
         var numVal = parseFloat(val);
         if (!isNaN(numVal)) {
             if (limit.min !== "" && numVal < parseFloat(limit.min)) flag = "L";
             if (limit.max !== "" && numVal > parseFloat(limit.max)) flag = "H";
         }
      }
      rowsToInsert.push([orderId, tName, pName, val, flag, timestamp, user]);
    }

    // ບັນທຶກລົງ Sheet ບາດດຽວ (ແຖວໃໝ່ທັງໝົດ)
    if (rowsToInsert.length > 0) {
        resSheet.getRange(resSheet.getLastRow() + 1, 1, rowsToInsert.length, rowsToInsert[0].length).setValues(rowsToInsert);
    }

    // ປ່ຽນສະຖານະບິນເປັນ Completed
    var orderSheet = ss.getSheetByName('Test_Orders');
    var orderData = orderSheet.getDataRange().getValues();
    for (var o = 1; o < orderData.length; o++) {
      if (String(orderData[o][0]).trim() === String(orderId).trim()) {
          orderSheet.getRange(o + 1, 19).setValue("Completed");
          break;
      }
    }
    logActivity(user, "Enter/Edit Result", orderId, "ບັນທຶກ/ແກ້ໄຂ ຜົນກວດສຳເລັດ");
    return { success: true, message: "ບັນທຶກຜົນກວດສຳເລັດແລ້ວ!" };
  } catch(e) { return { success: false, message: e.toString() }; }
}

function getSavedResults(orderId) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var resSheet = ss.getSheetByName('Test_Results');
    if (!resSheet) return [];
    var resData = resSheet.getDataRange().getValues();

    var paramSheet = ss.getSheetByName('Test_Parameters');
    var paramData = paramSheet ? paramSheet.getDataRange().getValues() : [];

    var paramLookup = {};
    for(var p=1; p<paramData.length; p++) {
       var key = paramData[p][0] + "_" + paramData[p][1]; 
       paramLookup[key] = {
         unit: paramData[p][4],
         min: paramData[p][5],
         max: paramData[p][6]
       };
    }

    var savedResults = [];
    for (var i = 1; i < resData.length; i++) {
      if (resData[i][0] === orderId) {
        var tName = resData[i][1];
        var pName = resData[i][2];
        var lookupKey = tName + "_" + pName;
        var pInfo = paramLookup[lookupKey] || {unit: '', min: '', max: ''};
        
        var nRange = "-";
        if (pInfo.min !== '' && pInfo.max !== '') {
            nRange = pInfo.min + " - " + pInfo.max;
        }

        savedResults.push({
          testName: tName,
          paramName: pName,
          value: resData[i][3],
          flag: resData[i][4], 
          unit: pInfo.unit,
          normalRange: nRange
        });
      }
    }
    return savedResults;
  } catch(e) {
    return [];
  }
}
